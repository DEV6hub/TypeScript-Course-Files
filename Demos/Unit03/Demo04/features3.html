let wait1000: Promise = new Promise((resolve: any, reject: any) => {
  setTimeout(resolve, 1000);
});

wait1000.then(() => console.log('Yay!'));


function someAsyncThing(): Promise {
  return new Promise((resolve: any, reject: any) => {
    resolve();
  });
};

function someOtherAsyncThing(): Promise {
  return new Promise((resolve: any, reject: any) => {
    reject('something went wrong');
  });
};

someAsyncThing()
  .then(() => someOtherAsyncThing())
  .catch((error: string) => console.log('oh no', error));
